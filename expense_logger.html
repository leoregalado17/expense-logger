Connect to Google Sheets
1
Go to sheets.new and create a new spreadsheet. Name it Budget 2026.
2
Click Extensions → Apps Script. Delete everything in the editor and paste the script below.
function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheetName = data.month || "Expenses"; var sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); sheet.appendRow([ "Date","Category","Description", "Amount","Payment","Location","Notes" ]); sheet.getRange(1,1,1,7).setFontWeight("bold"); } sheet.appendRow([ data.date, data.category, data.description, data.amount, data.payment, data.location, data.notes ]); return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:"error",message:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
3
Click Deploy → New deployment. Set type to Web app, set "Who has access" to Anyone, click Deploy. Copy the Web App URL it gives you.
4
Paste your Web App URL below and tap Save.
